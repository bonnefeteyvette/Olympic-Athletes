# Filter non-null medal records
medal_df = df[df['Medal'].notna()]

# Group by NOC and Medal, count occurrences
noc_medal_counts = medal_df.groupby(['NOC_y', 'Medal']).size().unstack().fillna(0)

# Select top 10 NOCs by total medals
top_noc = noc_medal_counts.sum(axis=1).sort_values(ascending=False).head(10).index
top_noc_medals = noc_medal_counts.loc[top_noc]

# Plot
top_noc_medals.plot(kind='bar', stacked=True, figsize=(12, 6), colormap='tab20c')
plt.title("Medal Distribution by Top 10 NOCs")
plt.xlabel("NOC")
plt.ylabel("Number of Medals")
plt.legend(title="Medal Type", bbox_to_anchor=(1.05, 1))
plt.tight_layout()
plt.show()
